## 使用示例

```typescript
import { OutcomeScheduler } from './modules/outcome-scheduler';

// 创建调度器
const scheduler = new OutcomeScheduler();

// 注册每日报告任务
scheduler.templateEngine.registerTemplate('daily-report',
  'Daily report generated at {{executionTime}}\n\n{{summary}}'
);

const job = {
  id: 'daily-report',
  name: 'Daily Report',
  schedule: {
    type: 'cron',
    expression: '0 9 * * *', // 每天9点
    enabled: true
  },
  template: 'daily-report',
  defaultVariables: {
    summary: 'All systems normal'
  },
  channels: [
    {
      type: 'feishu',
      destination: 'https://open.feishu.cn/open-apis/bot/v2/hook/xxx',
      title: 'Daily Report'
    }
  ],
  retry: {
    maxRetries: 3,
    retryIntervalMs: 1000,
    exponentialBackoff: true
  },
  costPerExecution: 10, // 10 cents per execution
  needRender: true
};

scheduler.addJob(job);

// 启动调度
scheduler.start();

// 手动触发
const result = await scheduler.triggerJob('daily-report');
console.log(result.success);
console.log(scheduler.getBillingSummary());
```
